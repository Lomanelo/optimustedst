# SMTP Configuration for Email Functionality
# Configure these with your email provider settings
# Copy this file to .env.local and update with your actual values

# SMTP Host (e.g., smtp.gmail.com, smtp.outlook.com, smtp.mailtrap.io)
SMTP_HOST=smtp.gmail.com

# SMTP Port (587 for TLS, 465 for SSL, 25 for non-secure)
SMTP_PORT=587

# SMTP Security (true for SSL on port 465, false for TLS on port 587)
SMTP_SECURE=false

# SMTP Authentication
SMTP_USER=your-email@gmail.com
SMTP_PASSWORD=your-app-password

# From Email Address (usually same as SMTP_USER)
SMTP_FROM_EMAIL=your-email@gmail.com

# Gmail App Password Setup Instructions:
# 1. Enable 2-factor authentication on your Gmail account
# 2. Go to Google Account settings > Security > App passwords
# 3. Generate an app password for "Mail"
# 4. Use the generated password (not your regular Gmail password)

# For other email providers:
# - Outlook/Hotmail: smtp.outlook.com, port 587
# - Yahoo: smtp.mail.yahoo.com, port 587
# - Custom SMTP: Contact your email provider for settings
